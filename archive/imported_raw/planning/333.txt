Minimal real code for the recurrence checker Here is a concrete checker skeleton you could drop in. from dataclasses import dataclass from sympy import sympify, simplify @dataclass(frozen=True) class SequenceCase: params: dict[str, int] value: int class RecurrenceChecker: def __init__(self, target_symbol: str): self.target_symbol = target_symbol def _lookup(self, cases_by_params, target: str, shifted_params: dict[str, int]): if target != self.target_symbol: raise ValueError(f"Unexpected target symbol {target}") key = tuple(sorted(shifted_params.items())) if key not in cases_by_params: raise KeyError(f"Missing case for {shifted_params}") return cases_by_params[key] def check(self, rule: RecurrenceRule, cases: list[SequenceCase]) -> tuple[bool, dict]: cases_by_params = { tuple(sorted(c.params.items())): c.value for c in cases } tested = [] for case in cases: env = dict(case.params) try: lhs_value = 0 for term in rule.lhs_terms: shifted = dict(env) for var, delta in term.shifts.items(): shifted[var] += delta coeff = sympify(term.coefficient_expr).subs(env) val = self._lookup(cases_by_params, term.target, shifted) lhs_value += coeff * val rhs_value = sympify(rule.rhs_expr).subs(env) residual = simplify(lhs_value - rhs_value) except KeyError: continue tested.append(case.params) if residual != 0: return False, { "failure_at": case.params, "residual": str(residual), "tested": tested, } return True, {"tested": tested} 15.
